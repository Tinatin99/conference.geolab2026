#define SPRING_K_1 200 // ზამბარის სიხისტე k1 #define MASS_M_1 0.10 // მასა m1 კგ-ში float positionX1 = 0.15; // საწყისი გადახრა (m) float velocityV1 = 0.0; // საწყისი სიჩქარე float damping = 0.05; // ჰაერის წინააღმდეგობის კოეფიციენტი b void setup() { Serial.begin(9600); } void loop() { // ჰუკის კანონი + დემპინგის ძალა: F = -k*x - b*v float springForce = -SPRING_K_1 * positionX1; float dampingForce = -damping * velocityV1; float totalForce = springForce + dampingForce; // ნიუტონის მე-2 კანონი: a = F / m float acceleration = totalForce / MASS_M_1; // ეულერის ინტეგრაცია (დროის ნაბიჯი dt = 0.05) velocityV1 += acceleration * 0.05; positionX1 += velocityV1 * 0.05; Serial.print("Position:"); Serial.println(positionX1); delay(50); }
