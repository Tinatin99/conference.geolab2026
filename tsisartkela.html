// სინათლის ტალღის სიგრძეები (nm) და ფერები const wavelengths = [700, 610, 570, 530, 470, 420]; const colors = ['#ff3b30', '#ff9500', '#ffcc00', '#4cd964', '#5ac8fa', '#5856d6']; function calculateRefraction(incomingAngle, waveLength, dispersion) { // გარდატეხის კოეფიციენტი წყლისთვის (~1.33) // დისპერსია იწვევს სხვადასხვა ფერის განსხვავებულ გარდატეხას const baseN = 1.33; const cauchyConst = 0.05; // კოეფიციენტი დამოკიდებულია ტალღის სიგრძეზე const n = baseN + cauchyConst * (1 / (waveLength / 500) - 1) * dispersion; // სნელიუსის კანონი: sin(r) = sin(i) / n return Math.asin(Math.sin(incomingAngle) / n); }
