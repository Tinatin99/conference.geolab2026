#include <LiquidCrystal_I2C.h> LiquidCrystal_I2C lcd(0x27, 16, 2); const int ledPin = 13; void setup() { pinMode(ledPin, OUTPUT); lcd.init(); lcd.backlight(); lcd.setCursor(0, 0); lcd.print("Wokwi Simulator!"); } void loop() { digitalWrite(ledPin, HIGH); // LED ჩართვა lcd.setCursor(0, 1); lcd.print("LED Status: ON "); delay(1000); digitalWrite(ledPin, LOW); // LED გამორთვა lcd.setCursor(0, 1); lcd.print("LED Status: OFF"); delay(1000); }
