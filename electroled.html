const int generatorPin = A0; // ანალოგური პინი const int ledPin = 9; // LED PWM პინი void setup() { pinMode(ledPin, OUTPUT); Serial.begin(9600); } void loop() { // ვკითხულობთ გენერირებულ ძაბვას (0 - 5V) int rawValue = analogRead(generatorPin); // ვახდენთ ძაბვის რუკირებას LED-ის სიკაშკაშეზე (0 - 255) int brightness = map(rawValue, 0, 1023, 0, 255); // ვანთებთ დიოდს შესაბამისი სიძლიერით analogWrite(ledPin, brightness); Serial.print("Voltage: "); Serial.print(rawValue * (5.0 / 1023.0)); Serial.println(" V"); delay(100); }
