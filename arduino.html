const int ldrPin = A0; // სინათლის სენსორი const int ledPin = 9; // LED ნათურა void setup() { pinMode(ledPin, OUTPUT); Serial.begin(9600); } void loop() { // ვკითხულობთ გარემო სინათლის დონეს (0-1023) int lightLevel = analogRead(ldrPin); // თუ ბნელა (მნიშვნელობა < 400), ვანთებთ LED-ს if (lightLevel < 400) { digitalWrite(ledPin, HIGH); } else { digitalWrite(ledPin, LOW); } delay(50); }
