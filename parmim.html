const float RHO = 1.72e-8; // სპილენძის კუთრი წინაღობა (Ohm*m) float calculateResistance(float L, float A) { // წინაღობა: R = ρ * (L / A) return RHO * (L / A); } float getCircuitCurrent(float V, float R1, float R2, bool isParallel) { float totalR = 0.0; if (isParallel) { // პარალელური ჩართვა: 1/R_total = 1/R1 + 1/R2 totalR = (R1 * R2) / (R1 + R2); } else { // მიმდევრობითი ჩართვა: R_total = R1 + R2 totalR = R1 + R2; } // ომის კანონი: I = V / R return V / totalR; } void loop() { float R1 = calculateResistance(2.0, 0.5e-6); // L=2მ, A=0.5მმ² float current = getCircuitCurrent(12.0, R1, R1, false); Serial.println(current); }
