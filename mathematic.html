const float PI_VAL = 3.14159; float currentX = 0.0; // რობოტის საწყისი X float currentY = 0.0; // რობოტის საწყისი Y void moveRobot(float angleDegree, float distance) { // კუთხის გადაყვანა რადიანებში float rad = angleDegree * (PI_VAL / 180.0); // კოორდინატების გამოთვლა ტრიგონომეტრიით currentX += distance * cos(rad); currentY += distance * sin(rad); Serial.print("Robot Position: ("); Serial.print(currentX); Serial.print(", "); Serial.print(currentY); Serial.println(")"); }
